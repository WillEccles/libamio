68 static int set_interface_attribs(int fd, UART_BAUD speed) {
69  struct termios tty;
70 
71  if (tcgetattr(fd, &tty) < 0) {
72  return -1;
73  }
74 
75  speed_t baud;
76  if (speed > UART_115200) {
77  baud = (speed_t)speed;
78  } else {
79  baud = BAUDS[speed];
80  }
81 
82  cfsetospeed(&tty, baud);
83  cfsetispeed(&tty, baud);
84 
85  tty.c_cflag |= (CLOCAL | CREAD); /* ignore modem controls */
86  tty.c_cflag &= ~CSIZE;
87  tty.c_cflag |= CS8; /* 8-bit characters */
88  tty.c_cflag &= ~PARENB; /* no parity bit */
89  tty.c_cflag &= ~CSTOPB; /* only need 1 stop bit */
90  tty.c_cflag &= ~CRTSCTS; /* no hardware flowcontrol */
91 
92  /* setup for non-canonical mode */
93  tty.c_iflag &= ~(IGNBRK | BRKINT | PARMRK | ISTRIP | INLCR | IGNCR | ICRNL | IXON);
94  tty.c_lflag &= ~(ECHO | ECHONL | ICANON | ISIG | IEXTEN);
95  tty.c_oflag &= ~OPOST;
96 
97  /* fetch bytes as they become available */
98  tty.c_cc[VMIN] = 0; // see https://stackoverflow.com/a/6947758/2712525
99  tty.c_cc[VTIME] = 5;
100 
101  if (tcsetattr(fd, TCSANOW, &tty) != 0) {
102  return -1;
103  }
104  return 0;
105 }
